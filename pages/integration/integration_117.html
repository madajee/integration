Greeting API

RAML

                
                    
greetinglib.raml

                    #%RAML 1.0 Library
                    types:
                      greet:
                        type: string
                        enum:
                          - hello
                          - good morning
                      lang:
                        type: string
                        enum:
                          - en
                          - es
                          - hi
                                    
            
                
                    
spanish-hello.raml

                    #%RAML 1.0 NamedExample
                    {
                        greeting: hola,
                        language: Spanish
                    }
                                    
            
                
                    
greeting-api.raml

                    #%RAML 1.0
                    title: greeting-api

                    uses:
                    greetingLib: datatypes/greetinglib.raml

                    /greeting:
                    get:
                        description: greeting
                        queryParameters:
                        greet:
                            description: greeting
                            type: greetingLib.greet
                            required: false
                        lang:
                            description: language
                            type: greetingLib.lang
                            required: false
                        responses:
                        200:
                            body:
                            application/json:
                                type: object
                                examples:
                                spanish-hello: !include examples/spanish-hello.raml
